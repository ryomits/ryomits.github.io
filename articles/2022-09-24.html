Go の定数定義は値を指定しないと前の値が使われるらしい。

package main

import "fmt"

const (
  a int32 = 1
  b
  c
)

func main() {
  fmt.Println(a)
  fmt.Println(b)
  fmt.Println(c)
}

% go run a.go
1
1
1

iota を使うと連番を作れるらしい。次の const で連番はリセット。

package main

import "fmt"

type Status int32

const (
  StatusUndefined Status = iota
  StatusActive
  StatusInActive
)

func main() {
  fmt.Println(StatusUndefined)
  fmt.Println(StatusActive)
  fmt.Println(StatusInActive)
}

% go run a.go
0
1
2
